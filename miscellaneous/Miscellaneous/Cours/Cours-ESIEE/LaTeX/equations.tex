\section{Single machine scheduling with sum of completion times}

MIP model:

\setcounter{equation}{0}
\begin{alignat}{2}
&\min \quad \sum \limits_{i \in [1,n]} s_{i}+d_{i} & \\
&s_i + d_i \leq s_j  & \forall (i,j) \in P \\
&s_i + d_i \leq s_j  + M.b_{ji} \quad & \forall i,j \in [1,n]: i \neq j \\
&b_{ij} = 1 - b_{ji}  & \forall i,j \in [1,n]: i < j \\
&\cpoInteger\ s_i \in \mathbb{Z}^+ & \forall i \in [1,n] \\
&\cpoInteger\ b_{ij} \in \{0,1\} & \forall i,j \in [1,n]: i \neq j
\end{alignat}

CP Optimizer model:

\setcounter{equation}{0}
\begin{alignat}{2}
&\min \quad \sum \limits_{i \in [1,n]} \cpoEndOf(o_{i}) & \\
&\cpoEndBeforeStart(o_i,o_j) \quad & \forall (i,j) \in P \\
&\cpoNoOverlap([o_i]_{i \in [1,n]}) &  \\
&\cpoInterval\ o_i\ \cpoSize\ d_i & \forall i \in [1,n]
\end{alignat}
