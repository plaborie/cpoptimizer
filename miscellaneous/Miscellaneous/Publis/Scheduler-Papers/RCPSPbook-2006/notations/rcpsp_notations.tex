\begin{itemize}
\item Activities
\begin{itemize}
\item $\cA=\{A_1,\ldots,A_n\}$ the set of activities/tasks, and 
\item dummy activities: $A_0$ (source/start of the project) and $A_{n+1}$ (sink/end of the project)
\item $V=\{A_0,A_{n+1}\}\cup\cA$
\item $p_i$ processing time/duration of activity $A_i$
\item $G(V,E,p)$ precedence graph: if $(A_i,A_j)\in E$ then $S_j\ge S_i+p_i$
\item $\Gamma_i\subseteq V$ set of direct successors of activity $A_i$
\item $\Gamma^{-1}_i\subseteq V$ set of direct predecessors of activity $A_i$
\item $S_i$ starting time of activity $A_i$,
\item $C_i$ finishing time of activity $A_i$,
\item  $ES_i$ earliest starting time (when computed during resolution)
\item $r_i$ release date (when part of input data), 
\item $EC_i$ earliest completion time,
\item $LS_i$ latest starting time, 
\item $LC_i$ latest completion time (when computed during resolution), 
\item $d_i$ due date (when part of input data)
\item $\delta_{ij}$ minimal (start-start-)distance between $A_i$ and $A_j$: $S_j-S_i\ge \delta_{ij}$
\item $S=\{S_0,S_1,\ldots,S_{n+1})$ a schedule
\item $C_{\max}$, $L_{\max}$: makespan and maximum lateness
\item $L_i$, $T_i$ : lateness and tardiness
\end{itemize}
\item Time
  \begin{itemize}
\item $H$ time horizon: $S_0=0$ and $S_{n+1}\le H$
%\item $\cT=\{0,1,\ldots,H\}$ the set of time periods $t$
%\item $\cA_t=\{A_i\in\cA\ | S_i \le t < S_i+p_i\}$ the set of activities in process at time $t$
\item $\cT=\{0,1,\ldots,H-1\}$ the set of time periods $t$
\item $\cA_t=\{A_i\in\cA\ | S_i \le t < S_i+p_i\}$ the set of activities in process at period $t$
\item $\cS_H$ the set of feasible schedules with makespan not greater than $H$ 
\end{itemize}
\item Resources
\begin{itemize}
\item $\cR=\{R_1,\ldots,R_q\}$ the set of resources
\item $B_k$ capacity of resource $R_k$
\item $b_{ik}$ amount of resource $R_k$ used during the execution of $A_i$
\end{itemize}
\item Sequencing constraints and related sets
\begin{itemize}
\item $i-j$ short notation for constraint ($S_{i}+p_{i}\leq S_j$ or $S_{j}+p_{j}\leq S_i$): $i$ and $j$ are in disjunction.
\item $i\rightarrow j$ short notation for constraint $S_{i}+p_{i}\geq S_j$: $i$ precedes $j$. 
\item $i||j$ short notation for constraint ($S_{i}+p_{i}> S_j$ and $S_{j}+p_{j}> S_i$): $i$ and $j$ are in parallel.
\item $D$ a set of disjunctions $i-j$.
\item $\Omega$ a set (or clique) of activities pairwise in disjunction.
\item $\cF$: $l\in\cF$ is a feasible set of activites, i.e. a set of activities that can be all in parallel in a feasible solution.
\item $\cCm$: $C\in\cCm$ is a minimal critical (or forbidden) set of activites ($C$ is not a feasible set and any subset of $C$ is a feasible set).
\end{itemize}
\item Linear Programming
\begin{itemize}
\item $x_{ij}$ 0-1 variable: $x_{ij}=1\Leftrightarrow i\rightarrow j$
\item $f_{ijk}$ flow of resource $R_k$ units from $A_i$ to $A_j$
\item $y_{it}$ 0-1 variable: $y_{it}=1\Leftrightarrow S_i=t$
\item $z_{lt}$ 0-1 variable: $z_{lt}=1\Leftrightarrow$ feasible set $l\in\cF$ is in process at time $t$  
\end{itemize}
\end{itemize}
